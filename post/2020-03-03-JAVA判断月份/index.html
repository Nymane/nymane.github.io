package work01;

import java.util.Scanner;

public class Example04 {
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
//创建对象
System.out.println("输入一个1-12范围的整数:");
int i= sc.nextInt();
//使用输入对象从系统输入缓存区中读取数据
if(i1||i2||i12)
//关系运算符优先级高于逻辑运算符
System.out.println(i+ "月，属于冬季");
else if(i9||i10||i11)
System.out.println(i+"月，属于秋季");
else if(i6||i7||i8)
System.out.println(i+"月，属于夏季");
else if(i3||i4||i5)
System.out.println(i+"月，属于春季");
else
System.out.println("i="+i+",超出了1-12月的范围");
System.out.println("程序执行结束");
sc.close();
}
}
