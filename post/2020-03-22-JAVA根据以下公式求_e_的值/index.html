JAVA根据以下公式求 e 的值

java 编程
2020-03-22 
    package work01;

public class Example16 {
public static void main(String[] args) {
int n = 1;
long l = 1;
double d = 1, sum = 0;
while(d >= 10E-4) {
sum += d; //sum迭代求和
l *= n; //求 n！
d = 1.0 / l; //构造 单项式
n++; //n自增
}
System.out.print("d = " + d + ", sum = " + sum);
}

}
